df = pd.read_csv('KCPD_Crime_Data_2016.csv')

top_10 = df.Description.value_counts()[:10]
top_10.iplot(kind='bar', filename='kcpd-top10-crimes', 
             title='Top 10 Crimes In 2016', yTitle='Reported Incidents')

locations = []
for index, row in df.iterrows():
    lattitude = float(row['Latitude'])
    longitude = float(row['Longitude'])
    if lattitude != 0 and longitude != 0:
        locations.append((lattitude, longitude))

heatmap = gmaps.heatmap_layer(locations)
heatmap.max_intensity = 100
heatmap.point_radius = 5

fig = gmaps.figure()
fig.add_layer(heatmap)
